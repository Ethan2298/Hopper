private renderAnnotatedText(container: HTMLElement, text: string) {
  //* Walk each [[phrase||code]] annotation in text, emitting plain
  //* text, tooltip spans, and trailing text in order.
  let lastIndex = 0
  const regex = new RegExp(ANNOTATION_RE.source, ANNOTATION_RE.flags)

  for (const match of text.matchAll(regex)) {
    const phrase = match[1]
    const code = match[2]
    const start = match.index!
    const end = start + match[0].length

    //* Flush any plain text that sits between the last
    //* annotation and this one.
    if (start > lastIndex) {
      this.renderWithFileLinks(container, text.slice(lastIndex, start))
    }

    //* If the annotation has no code payload, treat it as
    //* ordinary text — no highlight, no tooltip.
    if (!code.trim()) {
      container.appendChild(document.createTextNode(phrase))
    } else {
      //* Otherwise, render the phrase as a hoverable span
      //* that reveals the code snippet on mouse-enter.
      const span = document.createElement("span")
      span.className = "book-code-highlight"
      span.textContent = phrase
      span.addEventListener("mouseenter", () => this.showCodeTooltip(span, code))
      span.addEventListener("mouseleave", () => this.hideCodeTooltip())
      container.appendChild(span)
    }

    lastIndex = end
  }

  //* Emit any trailing text that follows the final annotation.
  if (lastIndex < text.length) {
    this.renderWithFileLinks(container, text.slice(lastIndex))
  }
}

A · Paper-faithful

  • Literally what Google ships in the paper
  • Prose reads as "source of truth" — prose and code share the same visual plane
  • Simplest possible implementation (C-style buffer splice)
  • //* prefix is hard to miss — clearly AI-generated

B · Hopper-native

  • Feels like a book, not a commented source file
  • Consistent with existing Hopper reader aesthetic
  • Prose can use rich typography (italic, serif, larger size)
  • Decoration-based — doesn't mutate the buffer
A

Go paper-faithful

Render prose as //* comments. Closest to Google. Implement via approach C (buffer splice).

B

Go Hopper-native

Distinct prose blocks via CodeMirror widgets. Paper-inspired, not paper-copying.

C

Hybrid

Prose lives as //* comments in model output, but rendered with richer styling (italic, color) via CodeMirror decorations. Format = paper, rendering = Hopper.
